Following code inside `Dockerfile`

~~~docker
FROM node:12 // Layer 1

WORDDIR /app // Layer 2

COPY package*.json ./ // Layer 3

RUN npm install // Layer 4 [SHELL FORM]

COPY . . // Layer 5

ENV PORT=8080

EXPOSE 8080

CMD ["npm", "start"] // EXEC FORM


~~~
